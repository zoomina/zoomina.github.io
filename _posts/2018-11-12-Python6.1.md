#### 예제 1.
최대값과 최소값을 구하는 함수를 작성하세요.  

```python
def main():
  nlist = [int(n) for n in input('nlist 입력:').split()]

  print('최대값:', maxNum(nlist))
  print('최소값:', minNum(nlist))
```

<br>

**예제 답안**  
```python
import math

def maxNum(nlist):
    # 음의 무한대 대입
    nmax = -math.inf
    for n in nlist:
        # 비교해서 큰 수가 나오면 대입
        if n > nmax:
            nmax = n
    return nmax

def minNum(nlist):
    # 양의 무한대 대입
    nmin = math.inf
    for n in nlist:
        # 비교해서 작은 수가 나오면 대입
        if n < nmin:
            nmin = n
    return nmin

def main():
  nlist = [int(n) for n in input('nlist 입력:').split()]

  print('최대값:', maxNum(nlist))
  print('최소값:', minNum(nlist))

if __name__ == '__main__':
    main()
```

#### 예제 2.
약수의 개수를 구하는 프로그램을 작성하세요.  

```python
def main():
    number = int(input())
    print(numDivisor(number))
```

<br>

**예제 답안**

```python
def numDivisor(n):
    # 초기 count를 0으로 놓는다
    divisorNum = 0
    # 1부터 n까지의 모든 수를 range()객체를 이용해 비교
    for i in range(1, n+1):
        # 약수를 카운트
        if n % i == 0:
            divisorNum += 1

    return divisorNum

def main():
    number = int(input())
    print(numDivisor(number))

if __name__ == "__main__":
    main()
```

**예제 답안**

```python
def inputInfo(info):
    # 이미 학생 정보가 입력된 경우 리스트를 비운다
    if info != None:  # python에서는 null이 아니라 None을 사용합니다.
        info = []
    
    # 학생 정보를 입력과 동시에 append
    info.append(input('이름:'))
    info.append(input('학번:'))
    info.append(input('이메일:'))

    return info

def showInfo(info):
    if info == None:
        print('자료가 없습니다.')
    else:
        # 저장된 리스트에 idx값으로 
        print('이름:', info[0])
        print('학번:', info[1])
        print('이메일:', info[2])

def main():
  studentInfo = []

  while(True):
    # lower는 모든 문자를 소문자로 만들어주는 함수입니다.
    case = input('(I)nput, (S)how, (Q)uit:').lower()
    if case == 'i':
      studentInfo = inputInfo(studentInfo)
    elif case == 's':
      showInfo(studentInfo)
    elif case == 'q':
      print('프로그램을 종료합니다.')
      break
    else:
      print('(error)잘못된 입력입니다.')
      continue

if __name__ == '__main__':
    main()
```
